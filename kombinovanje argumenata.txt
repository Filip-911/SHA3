#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char *argv[]) {

    if(argc == 1) {
        printf("ERROR: Expected at least 1 argument\n");
        return 0;
    }

    int i = 1;
    char *str = (char*)malloc(strlen(argv[i]));
    strcat(str, argv[i]);

    for(i = 2; i < argc; i++)
    {
        str = (char*)realloc(str, (strlen(argv[i])+1));
        strcat(str, " ");
        strcat(str, argv[i]);
    }
    str+=3;
    printf("%s\n", str);
    free(str);
    return 0;

}
